Python 共内置了 list、 tuple 、dict 和 set 四种基本集合，每个集合对象都能够迭代
# tuple(元组)
tup = ('python', 2.7, 64)
for i in tup:
    print(i)
    
# dictionary
dic = {}
d={'a':1,'ga':s}
print(d['a'])
del d['a']
d['s']=10
dic['lan'] = 'python'
dic['version'] = 2.7
dic['platform'] = 64
for key in dic:
    print(key, dic[key])
